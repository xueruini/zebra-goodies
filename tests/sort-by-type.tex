% Regression: [sort=type] option groups the list-of-notes by type rather
% than by document order.  Multiple note types must each appear in their
% own group in the end-of-document summary; the document must compile
% without errors or spurious zebra warnings.
\documentclass{article}
\usepackage[sort=type]{zebra}

\begin{document}

\section{Introduction}

\note[alice]{first note}
\todo[bob]{second item}
\fixed[carol]{third item}
\comment{fourth item}
\note[dave]{fifth note}
\todo{sixth item}
\placeholder{seventh item}

\section{Background}

\comment[eve]{add more background}
\fixed{another fix}
\note{closing remark}

\end{document}
